/* eslint-disable no-unused-vars */

import {
  initializeAuth,
  getReactNativePersistence,
  browserLocalPersistence
} from 'firebase/auth';
import { initializeApp,  } from 'firebase/app';
import { initializeFirestore, getFirestore } from "firebase/firestore";
import ReactNativeAsyncStorage from '@react-native-async-storage/async-storage';
import { Platform } from "react-native";
import { getStorage } from "firebase/storage";
import Constants from "expo-constants";

/**
 * Firebase configuration and service initialization.
 *
 * This module initializes Firebase for the application and exports
 * configured instances of:
 *  - Firebase App
 *  - Firestore (with React Native–compatible settings)
 *  - Firebase Authentication (platform-aware persistence)
 *  - Firebase Storage
 *
 * It supports all three platforms:
 *  - iOS
 *  - Android
 *  - Web
 */

/**
 * Firebase project configuration.
 * Values are provided via Expo configuration or environment variables.
 */

// Initialize Firebase
const firebaseConfig = {
  apiKey: Constants.expoConfig.extra.firebaseApiKey,
  authDomain: "agas-shopping-list-demo.firebaseapp.com",
  projectId: "agas-shopping-list-demo",
  storageBucket: "agas-shopping-list-demo.firebasestorage.app",
  messagingSenderId: "326283031177",
  appId: "1:326283031177:web:614a2b367bbe8f39294c58"
};

/**
 * Initialize Firebase App.
 *
 * @type {import('firebase/app').FirebaseApp}
 */

const app = initializeApp(firebaseConfig);
// For more information on how to access Firebase in your project,
// see the Firebase documentation: https://firebase.google.com/docs/web/setup#access-firebase

/**
 * Initialize Firestore with React Native–compatible options.
 *
 * - `experimentalForceLongPolling` is required for React Native
 * - `useFetchStreams` is disabled to prevent networking issues
 */

initializeFirestore(app, {
  experimentalForceLongPolling: true,
  useFetchStreams: false,
});

/**
 * Firestore database instance.
 *
 * @type {import('firebase/firestore').Firestore}
 */

const db = getFirestore(app);

/**
 * Firebase Storage instance used for media uploads.
 *
 * @type {import('firebase/storage').FirebaseStorage}
 */

// Initialize Firebase Storage
const storage = getStorage(app);

/**
 * Firebase Authentication instance.
 *
 * Authentication persistence is platform-specific:
 *  - Web: browser local persistence
 *  - Native (iOS/Android): AsyncStorage persistence
 *
 * @type {import('firebase/auth').Auth}
 */

// code for supporting all three platforms: iOS, Android and Web. https://github.com/expo/fyi/blob/main/firebase-js-auth-setup.md
let auth;

if (Platform.OS === "web") {
  auth = initializeAuth(app, {
    persistence: browserLocalPersistence,
  });
}
else {
  auth = initializeAuth(app, {
  persistence: getReactNativePersistence(ReactNativeAsyncStorage)  
}); 
}


export { app, db, auth, storage };
